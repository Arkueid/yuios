static page_entry_t *get_pde()
{
    // 页目录存放在内存最高位
    return (page_entry_t *)(0xfffff000); // 线性地址
}

static page_entry_t *get_pte(u32 vaddr)
{
    return (page_entry_t *)(0xffc00000 | (DIDX(vaddr)) << 12);
}

{

    BMB;
